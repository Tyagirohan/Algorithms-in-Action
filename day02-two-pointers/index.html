Palindrome Check

function isPalindrome(str) {
    let left = 0;
    let right = str.length - 1;
    
    while (left < right) {
        if (str[left] !== str[right]) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}

Pair Sum Finder

function findPairs(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    const pairs = [];
    
    while (left < right) {
        const sum = arr[left] + arr[right];
        if (sum === target) {
            pairs.push([arr[left], arr[right]]);
            left++;
            right--;
        } else if (sum < target) {
            left++;
        } else {
            right--;
        }
    }
    return pairs;
}
